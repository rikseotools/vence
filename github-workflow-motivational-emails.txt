name: üìß Motivational Emails Scheduler
on:
  schedule:
    - cron: '0 */6 * * *'
  workflow_dispatch:
    inputs:
      reason:
        description: 'Test reason'
        required: false
        default: 'Manual trigger'

jobs:
  send-emails:
    runs-on: ubuntu-latest
    steps:
      - name: Send emails
        env:
          CRON_SECRET: ${{ secrets.CRON_SECRET }}
        run: |
          echo "üöÄ Starting email workflow..."
          echo "Secret configured: ${#CRON_SECRET} chars"
          
          echo "üìß Sending motivational emails..."
          response=$(curl -s -w "\n%{http_code}" \
            --request POST \
            --header "Authorization: Bearer ${CRON_SECRET}" \
            --header "Content-Type: application/json" \
            https://www.vence.es/api/emails/send-motivational-scheduled)
          
          http_code=$(echo "$response" | tail -n1)
          response_body=$(echo "$response" | head -n -1)
          
          echo "HTTP Status: $http_code"
          echo "Response: $response_body"
          
          if [ $http_code -eq 200 ]; then
            echo "‚úÖ Emails motivacionales enviados exitosamente"
          else
            echo "‚ùå Error enviando emails motivacionales"
            exit 1
          fi
